public JPanel getBar(int width, int height, int orientation){
JPanel p = new JPanel();
bar = new JProgressBar();
bar.setOrientation(orientation);
if (orientation == JProgressBar.VERTICAL)
    bar.setPreferredSize(new Dimension(height, width));
else bar.setPreferredSize(new Dimension(width, height));
bar.setStringPainted(true);
/* This line is valid only for non rounded corner */
bar.setBorder(BorderFactory.createEtchedBorder());
/* This line is invalid for UI*/
bar.setForeground(Color.red);
/* Set the Illusion UI*/
bar.setUI(ipbi = new IllusionProgressBarUI(bar));
/* if you want round corner UI, you can choose RoundNOROUND, RoundOVAL, RoundNOROUND or custum double[] {x,y}
ipbi.setRoundCorner(IllusionProgressBarCoord.RoundOVAL);
/* Set multicolor effect with array of Colors, any Color will split in darker and brighter for effect (Default: dark = rgba(48,143,252,100);light = rgba(145,217,255,100);)*/
ipbi.setDarkLightColors(col);
/* Set border thickness (Default=0)*/
ipbi.setBorder(1);
/* Set Color of string without bar in background (Default=BLACK)*/
ipbi.setSelectionBackground(Color.RED);
/* Set Color of string on the bar in foreground (Default=WHITE)*/
ipbi.setSelectionForeground(Color.YELLOW);
/* If you want shadow*/
ipbi.setShadowPainted(true);
/* If you want more quality, for now no effect or too poor effect*/
ipbi.setHighQuality(true);
if (orientation == JProgressBar.HORIZONTAL)
   p.setLayout(new FlowLayout(FlowLayout.CENTER,0,(lateralHeight -bar.getPreferredSize().height)/2));
p.add(bar);
return p;
}
